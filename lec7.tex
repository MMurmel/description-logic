\begin{lemma}
	For arbitrary concept descriptions $C$ and arbitrary TBoxes $ \mathcal{T}$ the following hold:
	\begin{itemize}
		\item $\left| \text{sub}(C) \right| \leq \text{size}(C)$ 
		\item $\left| \text{sub}(\mathcal{T}) \right| \leq \text{size}(\mathcal{T})$
	\end{itemize}
\end{lemma}
